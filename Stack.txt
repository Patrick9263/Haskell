append(value, targetList) =
  targetList ++ [value]
:}

:{
helper (n,x,values) =
  let
  n = length values
  firstElement = head values
  nextList = tail values
  in
  if n == 0
    then []
    else append(firstElement,(helper(n-1,x+1,nextList)))
:}

:{
helper2 stack =
  reverse (tail (reverse stack))
:}

:{
push(values) =
  helper(0,0,values)
:}


:{
pop(howMany, list) =
  if howMany <= 0
    then list
    else pop(howMany-1,helper2(list))
:}
