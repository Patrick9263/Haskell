append(value, targetList) =

:{
helper (n,x,values) =
  let
  n = length values
  firstElement = head values
  nextList = tail values
  in
  if n == 0
    then []
    else append(firstElement,(helper(n-1,x+1,nextList)))
:}

:{
helper2 stack =
  reverse (tail (reverse stack))
:}

:{
push(values, stack) =
  stack ++ helper(0,0,values)
:}

  if howMany <= 0
    then list
